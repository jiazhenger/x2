# 用法 1、返回给定数组（array）的平均数 2、如果数组为空，返回 undefined 3、此方法忽略无效值(如 NaN 和 undefined) # 语法 d3.mean(array, accessor) # 参数 array ： 必需，数组 accessor ： 可选，一个处理数据的函数 # 示例 # 无 accessor 函数 const r = d3.mean([ ]) // => undefined const r = d3.mean([ 1, 7, 10 ]) // => 6 # 有 accessor 函数，相当于 map( )，返回处理后的数组，再取平均数 const r = d3.mean([ 1, 7, 10 ],function(value, index, arr){ // => 16 return value + 10 })
