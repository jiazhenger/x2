# 用法 返回给定数组 numbers 的 p 分位数，其中p 是一个0到1范围的数 # 语法 d3.quantile(numbers, p) # 参数 numbers ： 必需，数字数组 p ： 必需，0 ~ 1 的数 # 示例 const r = d3.quantile([ 1, 7, 10 ]，0) // => 1 const r = d3.quantile([ 1, 7, 10 ]，0.5) // => 7 const r = d3.quantile([ 1, 7, 10 ]，0.2) // => 3.4000000000000004 const r = d3.quantile([ 1, 7, 10 ]，1) // => 10
