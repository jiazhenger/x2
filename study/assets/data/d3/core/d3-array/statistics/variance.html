# 用法 返回给定数组（array）的无偏总体方差 # 语法 d3.median(variance, accessor) # 参数 array ： 必需，数组 accessor ： 可选，一个处理数据的函数 # 示例 # 无 accessor 函数 const r = d3.variance([ ]) // => undefined const r = d3.variance([ 1, 7, 10 ]) // => 21 # 有 accessor 函数，相当于 map( )，返回处理后的数组，再取方差 const r = d3.variance([ 1, 7, 10 ],function(value, index, arr){ // => 21 return value + 10 })
