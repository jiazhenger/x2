# 用法 返回两个数组 a 和 b 的 Cartesian product(笛卡尔积) # 语法 d3.cross(a, b, reducer) # 参数 a ：必需，数组 b ：必需，数组 reducer ：可选，函数 # 示例 # 无函数 const arr = d3.cross([1, 2], ['x', 'y']) // => [[1,x],[1,y],[2,x],[2,y]] # 有函数 const arr = d3.cross([1, 2], ['x', 'y'],function(a, b){ // => ["1x", "1y", "2x", "2y"] return a + b })
