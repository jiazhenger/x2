# 用法 设置文本内容将取代任何现有的子元素 # 语法 transition.text(value) # 参数 value ：必需，文本内容（常量、函数） # 函数 this 指向当前 dom # 示例 # value 是常量：所有的元素被给予相同的属性值 transition.text('d3.js') # value 是函数：该函数为每个选定的元件（按顺序）计算，被传递的参数是当前数据d与当前索引i，用this表示当前的DOM元素 transition.text('style',function(data, index){ if( i%2 === 0 ){ return 'd3.js' // 返回值被用于设置每个元素的属性 }else{ return 'three.js' } }) # 清除内容 transition.text(null)
