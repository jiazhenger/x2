# 用法 为每个选中的元素的指定的事件 # 语法 transition.on(typenames, listener) # 参数 typenames ：必需，事件类型 listener ：可选，添加或移除一个监听函数 # typenames start ：过渡开始 end ：过渡结束 interrupt ：过渡被中断 # 示例 # 添加事件 d3.selectAll('div').transition( ) .on('start',function( ){ }) .on('end',function( ){ }) .on('interrupt',function( ){ }) # 移除事件 d3.selectAll('div').transition( ) .on('start', null) .on('end', null) .on('interrupt', null)
