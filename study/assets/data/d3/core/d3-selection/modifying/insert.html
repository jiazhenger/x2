# 用法 添加 dom 到元素内部 # 语法 selection.append(name,before) # 参数 name ：必需，插入的 html 元素（字符串）或 已存在的 node 元素（函数） before ：可选，插入位置（字符串、函数） :first-child : // 插入到子元素最前 :last-child : // 插入到子元素最后 :nth-child(3) : // 插入到第3个子元素 # 函数 this 指向当前 dom # 示例 # 插入 html 元素到子元素最后，同 append( ) selection.insert('div') # 插入 html 元素到指定位置 selection.insert('div',':last-child') # 函数计算插入元素 selection.insert(function(data,index){ return index % 2 === 0 ? document.querySelector('.class') : document.querySelector('#id') },':first-child') # 函数计算位置 selection.insert('div', function(data,index){ return this.childNodes[2] // 返回子元素的一个 node })
