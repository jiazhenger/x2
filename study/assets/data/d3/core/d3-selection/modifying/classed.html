# 用法 设置 class 样式 # 语法 selection.classed(name, value) # 参数 name ：必需，className（字符串）、样式判断对象（对象字面量） value ：必需，true：添加、false：删除（布尔值） # 示例 # 判断指定 className 是否存在，存在返回 true，不存在返回 false const bool = selection.classed('fx') # 新增 className selection.classed('fx', true) selection.classed('fx wh rel', true) # 移除 className selection.classed('fx', false) selection.classed('fx wh rel', false) # 新增移除 className setObj('classed') // 重写 classed selection.classed({ fx:true, wh:false, 'abs_lt':false })
