# 用法 设置属性 # 语法 selection.attr(name, value) # 参数 name ：必需，html 属性（字符串）、属性对象（对象字面量） value ：可选，属性值（字符串、函数） # 函数 this 指向当前 dom # 示例 # 获取属性 const href = selection.attr('href') const style = selection.attr('style') # 添加属性 selection.attr('href','http://www.x.com') selectionattr('style','color:red') # 函数添加属性 selection.attr('style',function(data,index){ return index % 2 === 0 ? 'color:red' : 'color:blue' }) # 添加多个属性 setObj('attr') // 重写 attr selection.attr({ id:'box', style:'color:red' })
