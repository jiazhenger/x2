# 用法 设置 style 样式 # 语法 selection.style(name, value) # 参数 name ：必需，css 样式名（字符串）、样式对象（对象字面量） value ：可选，属性值（字符串、函数） # 函数 this 指向当前 dom # 示例 # 添加样式 selection.style('color','red') # 函数添加属性 selection.style('color',function(data,index){ return index % 2 === 0 ? 'red' : 'blue' }) # 添加多个属性 setObj('style') // 重写 style selection.style({ color:'red', fontSize:'16px' })
