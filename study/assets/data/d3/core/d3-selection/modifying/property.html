# 用法 一些 HTML 元素具有特殊的属性，使用标准的属性或样式是不可寻址的，如：className、checked、disabled # 语法 selection.property(name, value) # 参数 name ：必需，属性名（字符串）、属性对象（对象字面量） value ：可选，属性值（字符串、布尔值、函数） # 函数 this 指向当前 dom # 示例 # 判断指定 property 的真假 const bool = selection.property('checked') const bool = selection.property('disabled') # 设置属性为 true selection.property('checked', true) selection.property('disabled', true) # 设置属性为 false selection.property('checked', false) selection.property('disabled', false) # 函数添加属性 selection.property('checked',function(data,index){ return index % 2 === 0 }) # 添加多个属性 setObj('property') // 重写 property selection.property({ checked:true, disabled:true })
