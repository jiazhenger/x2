# 用法 添加 dom 到元素内部 # 语法 selection.append(name) # 参数 name ：必需，要添加的 html 元素（字符串）或 已存在的 node 元素（函数） # 函数 this 指向当前 dom # 示例 # 添加 html 元素到子元素最后 selection.append('div') # 函数添加 html 元素 selection.append(function(data,index){ return index % 2 === 0 ? document.querySelector('.class') : document.querySelector('#id') })
