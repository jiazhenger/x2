import * as d3 from 'd3' # 用法 选择全部元素 # 语法 d3.selectAll(name) # 参数 name ：必需，要选择的 css 选择器（字符串）、node 数组（nodes） # 示例 # 字符串选择器 const selection = d3.selectAll('p') // html 选择器 const selection = d3.selectAll('.class') // class 选择器 # node 选择器。选择指定的元素数组 const selection = d3.select(document.links) const selection = d3.select(this.childNodes) const selection = d3.select(document.querySelectorAll('.class'))
