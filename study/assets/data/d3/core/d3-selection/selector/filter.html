# 用法 过滤选择元素 # 语法 selection.filter(selector) # 参数 selector ：必需，css 选择器（字符串、函数） :first-child : // 第一个元素 :last-child : // 最后一个元素 :nth-child(even) : // 偶数 :nth-child(odd) : // 奇数 :nth-child(2n) : // 偶数 :nth-child(2n+1) : // 奇数 :nth-child(3) : // 第3个子元素 # 函数 this 指向当前 dom # 示例 # 字符串选择器 const selection = d3.selectAll('div').filter(:nth-child(even)) // 选择偶数 const selection = d3.selectAll('div').filter(:nth-child(odd)) // 选择奇数 # 函数 d3.selectAll('div').filter(function(data,index){ return ':nth-child(2)' })
