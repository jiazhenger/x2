import * as d3 from 'd3' # 用法 选择第一个元素 # 语法 d3.select(name) # 参数 name ：必需，要选择的 css 选择器（字符串）、node 元素（node） # 示例 # 字符串选择器 const selection = d3.select('p') // html 选择器 const selection = d3.select('#id') // id 选择器 const selection = d3.select('.class') // class 选择器 # node 选择器。选择指定的节点 const selection = d3.select(document.body) const selection = d3.select(document.querySelector('#id')) const selection = d3.select(document.getElementById('id')) # react 选择器 window.onload = ( ) => { const selection = d3.select(this.refs.refName) }
