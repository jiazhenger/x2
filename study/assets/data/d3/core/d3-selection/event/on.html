# 用法 事件监听 # 语法 selection.on(name, listener, capture) # 参数 name ：必需，事件名（字符串 listener ：必需，监听函数（函数、null capture ：可选，指定类型 # 函数 this 指向当前 dom # 示例 selection.on('click',function(data,index){ }) # 可选的命名空间 selection.on('click.foo', (d,i)=>{ }) selection.on('click.boo', (d,i)=>{ }) # 删除监听 selection.on('click', null)
