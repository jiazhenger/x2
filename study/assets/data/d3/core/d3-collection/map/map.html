# 用法 构建一个新的 map，如果指定参数 object，复制参数object对象内所有枚举属性到map中 # 语法 d3.map(object, key) # 参数 object ：必需，一个数组或对象类型数据 key ：可选，key 函数，用来计算数组里每个数值的键 # 示例 # 对象 const map = d3.map({ a:0, b: 1}) // => Map {$a: 0, $b: 1} # 数组 const map = d3.map([ 'a', 'b' ]) // => Map {$0: "a", $1: "b"} # 字符串 const map = d3.map('abc') // => Map {$0: "a", $1: "b", $2: "c"} # 数字 const map = d3.map(123) // => Map { } # key 函数：重新定义键名 const map = d3.map([ {name: 'foo'}, {name: 'bar'} ]) // => Map { $0: {name: "foo"}, $1: {name: "bar"}} const map = d3.map([ {name: 'foo'}, {name: 'bar'} ], function(value, index, arr){ // => Map { $foo: {name: "foo"}, $bar: {name: "bar"} } return value.name // 返回新的键名 })
