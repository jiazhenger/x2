# 用法 返回一个 a 和 b 两个字符串之间的字符串插值器 # 语法 const interpolate = d3.interpolateString(a, b)(t) # 参数 a ：必需，开始值 b ：em>必需，结束值 t ：区间 [ 0, 1 ] # 方式： # 无数字：返回第二个参数 const r = d3.interpolateString('a', 'b')(.5) // => b # 有数字 const r = d3.interpolateString('a1', 'b1')(.5) // => b1.5 const r = d3.interpolateString('a1', '2')(.5) // => 1.5
