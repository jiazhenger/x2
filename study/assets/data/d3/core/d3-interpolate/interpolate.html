# 用法 返回一个介于 a 和 b 之间的默认插值器 # 语法 const interpolate = d3.interpolate(a, b)(t) # 参数 a ：必需，开始值 b ：必需，结束值，此数据类型决定插值类型 t ：区间 [ 0, 1 ] # b 不同类型： # 颜色类型：返回 interpolateRgb 插值器 const r = d3.interpolate('#000', '#fff')(.5) // => rgb(128, 128, 128) # 数字：返回 interpolateNumber 插值器 const r = d3.interpolate(0, 10)(.5) // => 5 # 字符串：返回 interpolateString 插值器 const r = d3.interpolate('a', 'b')(.5) // => 始终是 b # 数组：返回 interpolateArray 插值器 const r = d3.interpolate([ 1, 2 ], [ 3, 4 ])(.5) // => [2, 3] # 对象：返回 interpolateObject 插值器 const r = d3.interpolate({ a: 0 }, { b: 1 })(.5) // => 始终是 { b: 1 }
