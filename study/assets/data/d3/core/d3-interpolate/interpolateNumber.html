# 用法 返回一个 a，b 两个数字之间的数字插值器 # 语法 const interpolate = d3.interpolateNumber(a, b)(t) # 参数 a ：必需，开始值 b ：em>必需，结束值 t ：区间 [ 0, 1 ] # 示例： const r = d3.interpolateNumber(0, 2)(.5) // => 1 const r = d3.interpolateNumber(1, 2)(.5) // => 1.5 const r = d3.interpolateNumber(2, 2)(.5) // => 2 const r = d3.interpolateNumber(2, 0)(.5) // => 1 const r = d3.interpolateNumber(2, 1)(.5) // => 1.5 const r = d3.interpolateNumber(2, 2)(.5) // => 2
