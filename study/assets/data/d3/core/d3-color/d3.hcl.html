# 用法 构建一个 HCL 颜色通道的颜色对象. 返回的颜色对象中包含 h, c 和 l 三个属性 # 语法 d3.hcl(h, c, l, opacity) d3.hcl(specifier) d3.hcl(color) # 参数 h ： 必需，h 值 c ： 必需，c 值 l ： 必需，l 值 opacity ： 可选，透明度，[0,1] specifier ： 必需，颜色字符串 color ： 必需，颜色字符串 # 示例 const r = d3.hcl(0,0,0,0.5) // => Hcl {h: 0, c: 0, l: 0, opacity: 0.5} const r = d3.hcl(red) // => Hcl {h: 40.85261277607024, c: 106.83899941284552, l: 54.29173376861782, opacity: 1} const r = d3.hcl(rgba(0,0,0,.5)) // => Hcl {h: NaN, c: NaN, l: 0, opacity: 0.5}}
