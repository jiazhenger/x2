# 用法 创建一个新的 Lab 灰色( a = b = 0) # 语法 d3.gray(specifier, opacity) # 参数 specifier ： 必需，颜色字符串 opacity ： 必需，透明度 # 示例 const r = d3.gray(0,0.5) // => Lab {l: 0, a: 0, b: 0, opacity: 0.5} const r = d3.gray(red) // => Lab {l: NaN, a: 0, b: 0, opacity: 1} const r = d3.gray(rgba(0,0,0,.5)) // => Lab {l: NaN, a: 0, b: 0, opacity: 1}
