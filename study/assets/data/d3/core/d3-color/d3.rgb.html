# 用法 构建一个 RGB 颜色通道的颜色对象. 返回的颜色对象中包含 r, g 和 b 三个属性 # 语法 d3.rgb(r, g, b, opacity) d3.rgb(specifier) d3.rgb(color) # 参数 r ： 必需，r 值 g ： 必需，g 值 b ： 必需，b 值 opacity ： 可选，透明度，[0,1] specifier ： 必需，颜色字符串 color ： 必需，颜色字符串 # 示例 const r = d3.rgb(0,0,0,0.5) // => Rgb {r: 0, g: 0, b: 0, opacity: 0.5} const r = d3.rgb(red) // => Rgb {r: 255, g: 0, b: 0, opacity: 1} const r = d3.rgb(rgba(0,0,0,.5)) // => Rgb {r: 0, g: 0, b: 0, opacity: 0.5}
