# 用法 构建一个 HSL 颜色通道的颜色对象. 返回的颜色对象中包含 h, s 和 l 三个属性 # 语法 d3.cubehelix(h, s, l, opacity) d3.cubehelix(specifier) d3.cubehelix(color) # 参数 h ： 必需，h 值 s ： 必需，s 值 l ： 必需，l 值 opacity ： 可选，透明度，[0,1] specifier ： 必需，颜色字符串 color ： 必需，颜色字符串 # 示例 const r = d3.cubehelix(0,0,0,0.5) // => Cubehelix {h: 0, s: 0, l: 0, opacity: 0.5} const r = d3.cubehelix(red) // => Cubehelix {h: 351.8102617708421, s: 1.9488976453722686, l: 0.2999994453152424, opacity: 1} const r = d3.cubehelix(rgba(0,0,0,.5)) // => Cubehelix {h: NaN, s: NaN, l: 0, opacity: 0.5}
