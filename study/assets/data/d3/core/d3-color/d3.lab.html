# 用法 构建一个 Lab 颜色通道的颜色对象. 返回的颜色对象中包含 l, a 和 b 三个属性 # 语法 d3.lab(l, a, b, opacity) d3.lab(specifier) d3.lab(color) # 参数 l ： 必需，l 值 a ： 必需，a 值 b ： 必需，b 值 opacity ： 可选，透明度，[0,1] specifier ： 必需，颜色字符串 color ： 必需，颜色字符串 # 示例 const r = d3.lab(0,0,0,0.5) // => Lab {l: 0, a: 0, b: 0, opacity: 0.5} const r = d3.lab(red) // => Lab {l: 54.29173376861782, a: 80.8124553179771, b: 69.88504032350531, opacity: 1} const r = d3.lab(rgba(0,0,0,.5)) // => Lab {l: 0, a: 0, b: 0, opacity: 0.5}
