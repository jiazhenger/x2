# 用法 构建一个 HSL 颜色通道的颜色对象. 返回的颜色对象中包含 h, s 和 l 三个属性 # 语法 d3.hsl(h, s, l, opacity) d3.hsl(specifier) d3.hsl(color) # 参数 h ： 必需，h 值 s ： 必需，s 值 l ： 必需，l 值 opacity ： 可选，透明度，[0,1] specifier ： 必需，颜色字符串 color ： 必需，颜色字符串 # 示例 const r = d3.hsl(0,0,0,0.5) // => Hsl {h: 0, s: 0, l: 0, opacity: 0.5} const r = d3.hsl(red) // => Hsl {h: 0, s: 1, l: 0.5, opacity: 1} const r = d3.hsl(rgba(0,0,0,.5)) // => Hsl {h: NaN, s: NaN, l: NaN, opacity: NaN}
