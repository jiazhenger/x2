const fs = require('fs') const path = require('path') # 中文字体配置：可从 C:\Windows\Fonts 复制 .ttf 格式的字体到 fonts const fonts = { Roboto: { normal: path.join(__dirname, '../core/fonts/STXIHEI.TTF'), bold: path.join(__dirname, '../core/fonts/STXIHEI.TTF'), italics: path.join(__dirname, '../core/fonts/STXIHEI.TTF'), bolditalics: path.join(__dirname, '../core/fonts/STXIHEI.TTF') }, Roboto: { normal: 'node_modules/pdfmake/fonts/STXIHEI.TTF', bold: 'node_modules/pdfmake/fonts/STXIHEI.TTF', italics:'node_modules/pdfmake/fonts/STXIHEI.TTF', bolditalics: 'node_modules/pdfmake/fonts/STXIHEI.TTF' }, } # 实例化对象并配置字体 const PdfPrinter = require('pdfmake/src/printer'); const printer = new PdfPrinter(fonts); # pdf 内容 const docDefinition = { header: '页头' content: '主体' content: '页脚' } # 创建 pdf const pdfDoc = printer.createPdfKitDocument(docDefinition) # 将生成的 pdf 存储到指定文件目录 pdfDoc.pipe(fs.createWriteStream(path.join(__dirname, '../pdf/report.pdf'))).on('finish', ( ) => { }) pdfDoc.end( ) // 结束创建
