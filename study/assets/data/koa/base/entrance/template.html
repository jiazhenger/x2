const Koa= require('koa') const path = require('path') const views = require('koa-views') // npm i koa-views ejs --only=dev const app = new Koa( ) # 使用模板引擎中间件 app.use( views( path.join(__dirname, './views'), // 模板路由 = 绝对路由 + '/views' { extension: 'ejs' // 模板引擎为 ejs } ) ) # 使用模板 app.use( async ctx => { await ctx.render( 'index', // 模板文件名：views/index.ejs { title: 'hello koa2' } // 传递到模板的参数：<%= title %> ) }) app.listen(3005)
