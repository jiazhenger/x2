const Koa= require('koa') const Router = require('koa-router') // npm i koa-router --only=dev const app = new Koa( ) const router = new Router( ) # get 路由：请求指定的资源 router.get('/', async (ctx,next) => { ctx.body = Hello Koa' }) # post 路由：发送数据给服务器 router.post('/index', async (ctx,next) => { ctx.body = Hello Koa' }) router.put( ) // 使用请求中的负载创建或者替换目标资源 router.head( ) // 请求资源的头部信息 router.delete( ) // 删除指定的资源 router.options( ) // 获取目的资源所支持的通信选项 router.patch( ) // 对资源进行部分修改 # 使用路由中间件，必须在路由的最后 app.use( router.routes( ) ).use( router.allowedMethods( ) ) # 启用 HTTP 服务 app.listen(3005)
