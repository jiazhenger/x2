# 语法 Iterable <Match> matchAsPrefix( String string, # 匹配字符串 [ int start = 0 ] # 开始位置 ) # 示例 Match m = ('123').matchAsPrefix('123456') String input = m.input // 原字符串 int start = m.start // 匹配字符串的开始位置 int end = m.end // 匹配字符串的结束位置 int group = m.group(0) // 获取匹配的字符串 => 123456 List<String> groups = m.groups([0]) // 获取匹配的字符串 int count = m.groupCount // group 的数量 int pattern = m.pattern // 返回匹配的正则表达式
