# 用法 查找字符串索引值。找到返回 index 的位置；反之，返回 -1 # 语法 int indexOf( Pattern pattern, # 匹配字符串，或正则表达式 [ int start ] # 开始位置 ) # 示例 int num = ('123').indexOf('1'); // => 0 int num = ('123').indexOf('2',2); // 从第2个位置开始查找字符串 int num = ('123').indexOf(RegExp(r'[A-Z][a-z]')); // 正则表达式 bool bool = ('123').indexOf('1') == -1; // 未找到 => false bool bool = ('123').indexOf('1') != -1; // 找到 => true
