# 语法 Uri u = Uri.parse( String url, // url 字符串 [ int start = 0 ] // 开始解析位置 [ int end ] // 结束解析位置 ) # 示例 Uri u = Uri.parse(https://test.com:8080/html5/index?age=10&name=20#fragment) bool hasScheme = u.hasScheme // 是否有协议 => true bool isScheme = u.isScheme('https') // 是否是指定值 => true String scheme = u.scheme // 协议 => https String host = u.host // 主机 => test.com String authority = u.authority // 主机 => test.com String origin = u.origin // 域名 => https://test.com bool hasPort = u.hasPort // 是否有端口 => true String port = u.port // 端口 => 8080 bool hasQuery = u.hasQuery // 是否有参数 => true String query = u.query // 可选参数 => age=10&name=20 Map<String String> queryParameters = u.queryParameters // 可选参数 => {age: 10, name: 20} Map<String List<String>> queryParameters = u.queryParametersAll // 可选参数 => {age: [10], name: [20]} bool hasFragment = u.hasFragment // 是否有锚点 => true String fragment = u.fragment // 锚点 => fragment Uri url = u.removeFragment( ) // 移除锚点 => bool hasEmptyPath = u.hasEmptyPath // 路径是否为空 => hasEmptyPath bool isAbsolute = u.isAbsolute // 是否是绝对路径 => false bool hasAbsolutePath = u.hasAbsolutePath // 是否有绝对路径 => false String path = u.path // => List<String>> pathSegments = u.pathSegments // 路径片段 => ['html5','index'] UriData data = u.data // => String userInfo = u.userInfo // => int hashCode = u.hashCode // => 954277904 Uri url = u.normalizePath( ) // => Uri url = u.replace( ) // => Uri url = u.resolveUri( ) // => String str = u.toFilePath( ) // => String str = u.toString( ) // =>
