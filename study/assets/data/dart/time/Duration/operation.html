# 定义 Duration A = Duration(days: 1, hours: 1, minutes: 1, seconds: 1, milliseconds: 1, microseconds: 1); // => 25:01:01.001001 Duration B = Duration(days: 2, hours: 2, minutes: 2, seconds: 2, milliseconds: 2, microseconds: 2); // => 50:02:02.002002 # 加 Duration duration = A + B; // => 75:03:03.003003 # 减 Duration duration = A - B; // => -25:01:01.001001 # 乘 Duration duration = A * 2; // => 50:02:02.002002 # 除 Duration duration = A ~/ 2; // => 25:01:01.001001 # 比较 bool duration = A > B; // => false bool duration = A < B; // => true # 取反 bool duration = -A ; // => -25:01:01.001001
