# 定义 Duration duration = Duration(days: 1, hours: 1, minutes: 1, seconds: 1, milliseconds: 1, microseconds: 1); // => 25:01:01.001001 # 获取具体时间数 int days = duration.inDays; // 天数 => 1 int hours = duration.inHours; // 小时数 => 25 int minutes = duration.inMinutes; // 分钟数 => 1501 int seconds = duration.inSeconds; // 秒数 => 90061 int milliseconds = duration.inMilliseconds; // 毫秒数 => 90061001 int microseconds = duration.inMicroseconds; // 微秒数 => 90061001001
