# 定义时间对象 DateTime A = new DateTime(2019, 10, 1); DateTime B = new DateTime(2019, 10, 10); # 获取时间差对象 Duration D = A.difference(B); // A 相比 B 相差对象 => -216:00:00.000000 Duration D2 = A.difference(B); # 获取具体时间数 int days = D.inDays; // 天数 => -9 int hours = D.inHours; // 小时数 => -216 int minutes = D.inMinutes; // 分钟数 => -12960 int seconds = D.inSeconds; // 秒数 => -777600 int milliseconds = D.inMilliseconds; // 毫秒数 => -777600000 int microseconds = D.inMicroseconds; // 微秒数 => -777600000000 # 其它属性 String str = D.toString( ); // 转为字符串 => -216:00:00.000000 Duration abs = D.abs( ); // 转为正数 => 216:00:00.000000 bool negative = D.isNegative; // 是否是负数 => true int compare = D.compareTo(D2); // 判断两个 Duration 数据是否相等 => 相等：0，不相等：-1 Type type = D.runtimeType; // 数据类型 => Duration int hash = D.hashCode; // hashCode => 150500584
