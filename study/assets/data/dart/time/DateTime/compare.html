# 定义时间对象 DateTime A = new DateTime(2019, 10, 1); DateTime B = new DateTime(2019, 10, 10); # 判断一个时间是否在一个时间之后 bool v = A.isAfter(B); // A 是否在 B 之后 => false # 判断一个时间是否在一个时间之前 bool v = A.isBefore(B); // A 是否在 B 之前 => true # 判断两个时间是否相等 bool v = A.isAtSameMomentAs(B); // A 是否等于 B => false # 对比两个时间 int v = A.compareTo(B); // 返回值： A==B：0、A<B：-1、A>B: 1
