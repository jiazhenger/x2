/** 一、概念： 1、mixin声明的类，无法被继承，可用 with 与 implements 使用接口 2、mixin声明的类，无法创建实例 3、on:固定子类使用的类型 */ class InterfaceA{ } # mixin 定义接口 mixin InterfaceB on MyClass{ } // 接口只能被 MyClass 继承使用，否则异常； mixin InterfaceB{ # 接口属性 int a; int a = 10; # 接口方法 void run( ){ } } # with 使用 mixin 定义的接口 class MyClass extends InterfaceA with InterfaceB{ # 必须实现接口中的公共属性 int a = 10; # 必须实现接口中的公共方法 void run( )( ){ } } # implements 使用 mixin 定义的接口 class MyClass implements InterfaceA,InterfaceB{ }
