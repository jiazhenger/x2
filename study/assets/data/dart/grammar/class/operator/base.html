class MyClass{ final int x; final int y; const MyClass(this.x, this.y); #　重写 + MyClass operator + (MyClass v) { return MyClass(v.x + x, v.y + y); } #　重写 － MyClass operator － (MyClass v) { return MyClass(v.x － x, v.y － y); // v.x => 第二个实例的参数; x => 第一个实例的参数 } } void main( ){ MyClass　a = MyClass(10,20); MyClass　b = MyClass(1,2); var x = (a + b).x; // => a.x + b.x = 10 + 1 = 11 var x = (a - b).x; // => b.x - a.x = 1 - 9 = -9 }
