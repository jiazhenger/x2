# 父类：被继承的类 class Animal{ # 继承公共属性 String name = 'Animal'; # 私有属性不能被继承 static String color = 'white'; # 继承公共方法 void eat( ) { print('Animal Eat'); } # 私有方法不能被继承 static void run( ) { } # 构造函数不能被继承 Animal( ) { } } # 子类：继承的类 class Cat extends Animal{ String myName; Cat(){ # super 代表父类 this.myName = super.name; // 直接访问父级类的公共属性 super.eat(); // 直接访问父级类的公共方法 } } void main(){ Cat cat = Cat( ) cat.name; // => Animal cat.myName; // => Animal cat.eat(); // => Animal Eat }
