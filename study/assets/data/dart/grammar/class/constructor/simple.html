class MyClass{ # 声明参数的 4 种方式 var a; final b; int c; final int d; # 空构造函数 MyClass( ); # 有序参数：实例化的参数必须与构造参数保持一致 MyClass( this.a, this.b, this.c ); # 无序参数：实例化的参数可以为空 MyClass({ this.a, this.b, this.c }); } void main( ){ # 空构造函数 MyCalss my = MyCalss( ); # 有序参数 MyCalss my = MyCalss( 10,20,30 ); # 无序参数 MyCalss my = MyCalss( a:10, b:20,c:30, ); // 注意 a,b,c属性名无引号包裹 }
