/** * 每个方法都有一个返回值，如果没有指定返回值，默认会在函数最后附加一个return null的语句 */ # 无返回值 myFn( ){ print(30); } // 等同 void myFn( ){ print(30); } dynamic x = myFn( ); // 使用 => null # 有返回值，不定义任何类型 myFn(){ return 10 } dynamic x = myFn( ); // 使用 => x = 10 # 有返回值，定义返回值类型 num myFn( ){ return 20; // 返回值为数字 } num x = myFn( ); // 使用 => x = 20
