# 无参数类型传递参数 myFn(a,b){ } myFn(10,'some'); # 定义传递参数类型 myFn(int a, int b){ } myFn(10,20); # 可选参数 myFn(int a, int b, [int c,int d,e]){ } # 给可选参数设置默认认参数 myFn(int a, int b, [int c=10,int d,e=20]){ } # 无序传参、默认参数、可选参数 myFn({ int a:10, int b, c:50 }){ print(a + b); } myFn(b:20); // => 30 myFn(b:20,a:30); // => 50
