# 删除一个元素，删除成功返回 true，反之，false bool remove( Object value // 删除值 ) bool v = set.remove(3); // 删除列表中第一个 value = 3 的元素 # 删除多个元素 void removeAll( Iterable<Object> elements // List、Set ); set.removeAll( [ 1,2 ] ); set.removeAll( { 1,2 } ); # 删除所有满足条件的元素 void removeWhere( bool test(E element) // (v)=>true、(v){ return true} ) set.removeWhere((value) => value == 3); // =3 的删除 # 删除所有不满足条件的元素 void retainWhere( bool test(E element) // (v)=>true、(v){ return true} ) set.retainWhere((value) => value == 3); // !=3 的删除，只留 3 # 删除多余元素 void retainAll( Iterable<Object> elements // List、Set ) var set = { 1,2,3 }; set.retainAll( [ 1,2,3,4,5 ] ); // => {1, 2, 3} set.retainAll( { 1,2,3,4,5 } ); // => {1, 2, 3} # 删除列表的所有元素 void list.clear( );
