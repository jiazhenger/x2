# 必须有长度，索引值必须存在 Object list[index] = value list[0] = 1 # 添加元素，固定长度列表不能使用 void add( E value // 任意类型值 ) list.add(1) # 添加元素，固定长度列表不能使用 void addAll( Iterable<E> iterable // List、Set ) list.addAll( [ 4,5,6 ] ); // => [0, 1, 2, 4, 5, 6] list.addAll( { 4,5,6 } ); // => [0, 1, 2, 4, 5, 6] # 增加一个元素到指定下标 void insert( int index, // 索引值、下标值 E element // 添加值 ) list.insert(0,123); # 增加一个列表到指定下标 void insertAll( int index, // 索引值、下标值 Iterable<E> iterable // List、Set ) list.insertAll(3,[40, 50])
