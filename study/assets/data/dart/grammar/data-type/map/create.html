# 定义映射 Map map = { } Map set = {'a':1,'b':2,'c':3} // 属性值必须是字符串，否则，就是一个变量 # 定义一个空映射 Map map = new Map( ) # 定义一个空映射 Set set = new Map.identity( ); # 创建一个包含所有元素的映射 factory Map.from( Map other // Map ) Set v = Map.from( {'a':1,'b':2} ); // => {a:1,b:2} # 从元素创建一个映射 factory Map.of( Map<K, V> other, // Map ) Map v = Map.of( {'a':1,'b':2} ); // => {a:1,b:2} # 创建一个不可修改的映射 factory Map.unmodifiable( Map other // Map ) Set v = Map.unmodifiable( {'a':1,'b':2} ); // => {a:1,b:2} # 将 source 转为 Map Map<K2, V2> castFrom<K, V, K2, V2>( Map<K, V> source // Map ) Map map = Map.castFrom({'a':1,'b':'2'}); // => {a:1,b:2} # 创建一个新映射并添加所有条目 factory Map.fromEntries( Iterable<MapEntry<K, V>> entries // Iterable ) // 例1 List list = List<MapEntry<String, int>>(); list.add(MapEntry('a',1)); list.add(MapEntry('b',2)); Iterable iterable = list.take(list.length); // (MapEntry(a: 1), MapEntry(b: 2)) Map map = Map.fromEntries(iterable); // => {a:1,b:2} // 例2 Map map = Map.fromEntries( [MapEntry('a',1),MapEntry('b',2)] ); // => {a:1,b:2} // 例3 List list = [MapEntry('a',1),MapEntry('b',2)] Map map = Map.fromEntries( [...list] ); // 不能直接使用 list => {a:1,b:2} # 重新键和值，并创建一个新的映射 factory Map.fromIterable( Iterable iterable // List、Set { K key(element), // key:(element)=>element V value(element) // value:(element)=>element } ) Map map = Map.fromIterable( // => {a1: 10, a2: 20} [1,2], key:(v) => 'a$v', value:(v) => v*10 ); Map map = Map.fromIterable( // => {a1: 10, a2: 20} {1,2}, key:(v) => 'a$v', value:(v) => v*10 ); # 将两个 List 或 Set，合并成 Map factory Map.fromIterables( Iterable<K> keys, // List、Set Iterable<V> values // List、Set ) Map map = Map.fromIterables({'a','b'},[1,2] ); // => {a: 1, b: 2}
