div{ background-image : repeating-linear-gradient( direction, // direction:用角度值指定渐变的方向（或角度）, color-stop1, // 用于指定渐变的起止颜色 color-stop2, ... ) # 不带方向渐变 repeating-linear-gradient(red, yellow, blue) // 垂直线性渐变 # 带方向渐变 repeating-linear-gradient(to right, red, blue) // 从左侧开始的线性渐变 == 90deg repeating-linear-gradient(to left, red, blue) // 从右侧开始的线性渐变 == 270deg repeating-linear-gradient(to top, red, blue) // 从底部开始的线性渐变 == 0deg repeating-linear-gradient(to bottom, red, blue) // 从顶部开始的线性渐变 == 180deg，默认 repeating-linear-gradient(to top left, red, blue) repeating-linear-gradient(to top right, red, blue) repeating-linear-gradient(to bottom left, red, blue) repeating-linear-gradient(to bottom right, red, blue) # 带角度 repeating-linear-gradient(45deg, red, blue) repeating-linear-gradient(185deg, red, blue) # 带起位置的颜色 repeating-linear-gradient(to top, red 0%, blue 50%) // 0%： 用百分比指定起止色位置 repeating-linear-gradient(45deg, red 100px, blue 100px) // 100px： 用长度指定起止色位置 }
