div{ transform : none // 不进行转换 # 平移 translate(10px,20px) // 2D 转换 translate(10px,20px,30px) // 3D 转换 translateX(10px) // X 轴转换 translateY(20px) // Y 轴转换 translateZ(30px) // Z 轴转换 # 缩放 scale(2) // 2D 缩放，XY 轴同等缩放 scale(2,5) // 2D 缩放，X、Y 轴不同缩放 scale3d(2,5,8) // 3D 缩放 scaleX(2) // X 轴缩放 scaleY(5) // Y 轴缩放 scaleZ(8) // Z 轴缩放 # 旋转 rotate(90deg) // 2D 旋转 rotate3d(x,y,z,90deg) // 3D 旋转 rotateX(90deg) // X 轴旋转 rotateY(90deg) // Y 轴旋转 rotateZ(90deg) // Z 轴旋转 # 倾斜 skew(45deg,90deg) // 2D 倾斜 skewX(90deg) // X 轴倾斜 skewY(90deg) // Y 轴倾斜 # 倾斜 matrix(n,n,n,n,n,n) // 2D 矩阵转换，6 个值 matrix3d(n,...) // 2D 矩阵转换，16 个值，4x4 矩阵 # 为 3D 转换元素定义透视视图，可实现图片抖动效果 perspective(10px) } # 复合使用 div{ transform: translate(10px) scale(2) rotate(90deg) skew(45deg,90deg) matrix(n,n,n,n,n,n) perspective(10px) }
