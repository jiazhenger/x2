# 用法 一个包含当前渲染环境(RenderingContext)的功能细节的对象 # 语法 const renderer.capabilities = { floatFragmentTextures : true // 环境是否支持 OES_texture_float扩展 floatVertexTextures : true // 如果 floatFragmentTextures 和 vertexTextures 都是 true， 则此值为true getMaxAnisotropy : ƒ getMaxAnisotropy() // 返回最大可用各向异性 getMaxPrecision : ƒ getMaxPrecision(precision) // 返回顶点着色器和片元着色器的最大可用精度 isWebGL2 : false // 是否是 WebGL2 logarithmicDepthBuffer : false // 如果 logarithmicDepthBuffer 在构造器中被设为true且 环境支持EXT_frag_depth扩展，则此值为true maxAttributes : 16 // gl.MAX_VERTEX_ATTRIBS 的值 maxCubemapSize : 16384 // gl.MAX_CUBE_MAP_TEXTURE_SIZE 的值，着色器可使用的立方体贴图纹理的最大宽度*高度 maxFragmentUniforms : 1024 // gl.MAX_FRAGMENT_UNIFORM_VECTORS 的值，片元着色器可使用的全局变量(uniforms)数量 maxSamples : 0 // maxTextureSize : 16384 // gl.MAX_TEXTURE_SIZE 的值，着色器可使用纹理的最大宽度*高度 maxTextures : 16 // gl.MAX_TEXTURE_IMAGE_UNITS 的值，着色器可使用的纹理数量 maxVaryings : 30 // gl.MAX_VARYING_VECTORS 的值，着色器可使用矢量的数量 maxVertexTextures : 16 // gl.MAX_VERTEX_TEXTURE_IMAGE_UNITS 的值，顶点着色器可使用的纹理数量 maxVertexUniforms : 4096 // gl.MAX_VERTEX_UNIFORM_VECTORS 的值，顶点着色器可使用的全局变量 precision : "highp" // 染器当前使用的着色器的精度 vertexTextures : true // 如果 maxVertexTextures 大于0，此值为true (即可以使用顶点纹理) }
