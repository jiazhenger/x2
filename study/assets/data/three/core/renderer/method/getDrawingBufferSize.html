# 用法 获取绘图缓冲区，以像素(px)为单位 # 语法 renderer.getDrawingBufferSize(target) # 参数 target ：必需（new Three.Vector2( )） # 示例 const renderer = new Three.WebGLRenderer( ) const Vector2 = renderer.getDrawingBufferSize( new Three.Vector2( ) ) // => Vector2 { x: 1920, y: 427, width: 1920, height: 427 } const x = Vector2.x // 获取 x 轴坐标 const y = Vector2.y // 获取 y 轴坐标 const width = Vector2.width // 获取宽 const height = Vector2.height // 获取高
