# 用法 获取裁剪区 # 语法 renderer.getScissor( target ) # 参数 target ：必需（Vector2、Vector3、Vector4） # 示例 const Vector2 = renderer.getScissor(new Three.Vector2( )) // => Vector2 { x: 0, y: 0 } const Vector3 = renderer.getScissor(new Three.Vector3( )) // => Vector3 { x: 0, y: 0, z: 1920 } const Vector4 = renderer.getScissor(new Three.Vector4( )) // => Vector4 { x: 0, y: 0, z:1920, w:427 }
