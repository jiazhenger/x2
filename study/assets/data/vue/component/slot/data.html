# 子组件 my-component <template> <slot name='header' :data='{ x: 1 }' :rows='{ x: 1 }' :index='1'></slot> <slot :data='{ x: 1 } :rows='{ x: 1 }' :index='1'></slot> // 默认插槽 </template> # 父组件 <template> <div> # 具象插槽必须放使用 template <my-component> <template v-slot : header = 'props'>{{ props.data.x }} </template> <template v-slot = 'props'> {{ props.data.x }} </template> <template v-slot : default = 'props'> {{ props.data.x }}</template> # 缩写 <template #header = 'props'> {{ props.data.x }}</template> <template # = 'props'> {{ props.data.x }} </template> <template #default = 'props'> {{ props.data.x }} </template> </my-component> # 只有默认插槽时，组件的标签才可以被当作插槽的模板来使 <my-component v-slot : default = 'props'> {{ props.data.x }} </my-component> <my-component v-slot= 'props'> {{ props.data.x }} </my-component> # 缩写 <my-component #default = 'props'> {{ props.data.x }} </my-component> <my-component #= 'props'> {{ props.data.x }} </my-component> // 注意：默认插槽的缩写语法不能和具名插槽混用，因为它会导致作用域不明确 </div> </template>
