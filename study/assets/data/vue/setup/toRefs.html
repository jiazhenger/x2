<template><button >{{ count }}</button></template> <script> import { toRefs, reactive } from 'vue' export default { setup( ){ # 完整写法 const obj = toRefs( reactive({ count:1 }) ) // 只能转换 reactive 包裹的数据 const { count } = obj // 将变暴露给模板 const run = ( ) => obj.count.value ++ # 简写 const { count }= toRefs( reactive({ count:1 }) ) // 声明并将变暴露给模板 const run = ( ) => count.value ++ return { count, run } } } </script>
