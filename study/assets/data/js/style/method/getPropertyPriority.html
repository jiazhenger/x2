# 用法 返回指定的 CSS 属性是否设置了 "important!" 优先级。如果返回 "important" 则表明设置了优先级，否则没有 # 语法 const priority = el.style.getPropertyPriority(propertyname) // => important, undefined, ' ' # 参数 propertyname: 必需。一个字符串，表示要检测的属性名 # 示例 const style = document.styleSheets[0].rules[0].style const style = document.querySelector('#id').style const priority = style.getPropertyPriority('color') // => important, undefined, ' '
