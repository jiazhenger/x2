# 用法 设置年份 # 语法 (new Date).setFullYear( year, month, day ) (new Date).setUTCFullYear( year, month, day ) # 参数 year : 必需。表示年份的四位整数。用本地时间表示 month : 可选。表示月份的数值，介于 0 ~ 11 之间。用本地时间表示 day : 可选。表示月中某一天的数值，介于 1 ~ 31 之间。用本地时间表示 # 示例 const date = new Date( ) date.setFullYear( 2020, 6, 20 ) date.setUTCFullYear( 2020, 6, 20 )
