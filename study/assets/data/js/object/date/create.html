# 用法 Date 对象用于处理日期和时间 # 语法 const myDate = new Date // 当前日期和时间 const myDate = new Date( ) // 当前日期和时间 const myDate = new Date( milliseconds ) // 毫秒或微秒 const myDate = new Date( dateString ) // 时间字符串，如：'2020/06/17 12:30:50'、2020-06-17 const myDate = new Date( year, month, day, hours, minutes, seconds, milliseconds) // 年、月、日、时、分、秒、毫秒 # 示例 const myDate = new Date( 1592323996272 ) const myDate = new Date( '2020-4-26' ) const myDate = new Date( '2020/4/26 12:30:50' ) const myDate = new Date( 2020, 6, 17, 0, 15, 20, 30) const myDate = new Date( '2020', '6', '17', '0', '15', '20', '30')
