# 用法 将 NodeList 对象转为数组 # 语法 const array = Array.prototype.slice.call(nodeList, startIndex, endIndex) const array = [ ].slice.call(arguments, startIndex, endIndex) const array = Array.prototype.slice.apply(nodeList, [startIndex, endIndex]) const array = [ ].slice.apply(arguments, [startIndex, endIndex]) # 参数 nodeList : 必需。nodeList 列表 startIndex : 可选。数字。开始索引。默认 0 endIndex : 可选。数字。结束索引 # 返回值 一个数组 # 示例 const array = Array.prototype.slice.call(document.querySelectorAll('div')) const array = [ ].slice.call(document.querySelectorAll('div'), 0, 1) const array = Array.prototype.slice.apply(document.querySelectorAll('div')) const array = [ ].slice.apply(document.querySelectorAll('div'), [0,1]) # 相当于 const array = document.querySelectorAll('div').slice( ) const array = document.querySelectorAll('div').slice(0, 1)
