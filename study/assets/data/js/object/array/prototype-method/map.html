# 用法 返回一个新数组 # 语法 const array = Array.prototype.map.call(array, function) const array = [ ].map.call(array, function) # 参数 array : 必需。数组 function : 必需。map 中的回调函数 # 返回值 原数组 # 示例 const array = Array.prototype.map.call(document.body.childNodes, (v,i)=> v) # 相当于 const array = document.body.childNodes.map((v,i)=> v)
