# 用法 存储函数（或函数名），每当 readyState 属性改变时，就会调用该函数 # 语法 xhr.onreadystatechange(function) # 参数 function : 必需。函数 # 示例 xhr.onreadystatechange(function( ){ if(xhr.readyState === 4 && xhr.status === 200){ const data = xhr.responseText } if(this.readyState === 4 && this.status === 200){ const data = this.responseText } })
