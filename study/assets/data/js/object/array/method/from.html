# 用法 用于通过拥有 length 属性的对象或可迭代的对象来返回一个数组 # 语法 const arr = Array.from(object, mapFunction, thisValue) # 参数 object: 必需。要转换为数组的对象 mapFunction: 可选。数组中每个元素要调用的函数 thisValue: 可选。映射函数(mapFunction)中的 this 对象 # 返回值 Array # 数组去重 const arr = Array.from(new Set([1,2,2,3])) // => [1,2,3] # map 功能 const arr = Array.from([1,2,3], (value, index) => value * 10) // => [10,20,30] # 构造数据 const arr = Array.from({ '0': 100, '1': 200, length: 10 }, (value, index) => { index }) // 构造 10 个对象数据 => [ {index:0}, {index:1}, ... ]
