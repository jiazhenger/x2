# 用法 用来判断一个数组是否包含一个指定的值，如果是返回 true，否则false # 语法 const bool = arrayObject.includes(searchElement, fromIndex) # 参数 searchElement : 必需。需要查找的元素值 fromIndex : 可选。从该索引处开始查找 searchElement。如果为负值，则按升序从 array.length + fromIndex 的索引开始搜索。默认为 0 # 示例 const arr = [ 1, 2, 3 ] const bool = arr.includes(2) // => true const bool = arr.includes(4) // => false
