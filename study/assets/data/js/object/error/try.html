# 语法 try{ return }catch(err){ // 如果 try 中产生错误，则在此输出该错误的信息 const name = err.name // 设置或返回一个错误名 const message = err.message // 设置或返回一个错误信息(字符串) const description = err.description // 返回错误描述 const fileName = err.fileName // 返回出错文件名 const lineNumber = err.lineNumber // 返回第几行 const columnNumber = err.columnNumber // 返回第几列 return } # err.name 的值 EvalError : eval() 函数产生的错误。 注意: 新版的 JavaScript 使用 SyntaxError 替代 EvalError RangeError : 数值超出规定的范围 ReferenceError : 非法引用 SyntaxError : 语法错误 TypeError : 类型错误 URIError : encodeURI() 函数产生的错误 # 示例 const getJson = str => { try{ return JSON.parse(str) }catch( ){ return str } }
