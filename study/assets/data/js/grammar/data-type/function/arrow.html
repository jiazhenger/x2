# 用法 1、引入箭头函数有两个方面的作用：更简短的函数并且不绑定this 2、比函数表达式更简洁，并且没有自己的this，arguments，super或new.target 3、箭头函数表达式更适用于那些本来需要匿名函数的地方，并且它不能用作构造函数 # 基本语法 ( param1, param2, ... ) => { statements } ( param1, param2, ... ) => statements === ( param1, param2, ... ) => { return statements } ( param1 ) => { statements } === param1 => { statements } // 当只有一个参数时，圆括号是可选的 ( ) => { statements } // 没有参数的函数应该写成一对圆括号 # 高级语法 param => ( { foo:1 }) // 加括号的函数体返回对象字面量表达式 ( param1, param2, ...rest ) => { statements } // 剩余参数 ( param=defaultValue ) => { statements } // 默认参数 # 参数说明 # 返回值 默认情况下，函数是返回 undefined 的。想要返回一个其他的值，函数必须通过一个 return 语句指定返回值 # 示例 var foo = ( ) => { } // 无参数 var foo = name => { } // 一个参数 var foo = (name,age) => { } // 多个参数 var foo = name => name // 返回值 var foo = name => { return name } // 返回值 var foo = name => true ? 1 : 0 // 三元运算返回 var foo = ( ) => ({ a: 0 }) // 返回值为对象字面量 var foo = ( name=10 ) => { } // 默认参数 # 剩余参数 var foo = ( a,...args ) => { console.log(args) // 返回由实参组成的一个数组，=> [1,2] } foo(0,1,2) # 匿名箭头函数 [ ].map(v=>v) [ ].forEach((v,i)=>{ }) # 换行 var foo = ( ) => ( 0 ) # call( )、apply( )、bind( ) 调用 由于箭头函数没有自己的this指针，通过 call() 或 apply() 方法调用一个函数时，只能传递参数（不能绑定this），他们的第一个参数会被忽略 this.m = 0 var foo = x => this.m + 10 foo.call( { m: 10 }, 10 ) // => 10 # 使用箭头函数作为方法 var obj = { start: ( ) => { } } # 优先解析 let callback callback = callback || function( ) { } // 正常 callback = callback || ( ) => { } // 会报语法错误 callback = callback || ( ( ) => { } ) // 正常 # 没有 this # 不绑定 arguments # 没有 new 操作符，即不能实例化对象 # 没有构造函数 # 没有 prototype # 不能使用 yield 关键字，即：不能用作函数生成器
