# 用法 从ECMAScript 2015开始，在对象初始器中引入了一种更简短定义方法的语法，这是一种把方法名直接赋给函数的简写方式 # 语法 var obj = { # 定义属性 property: 10 property: ( true ? 1 : 0 ) # 匿名函数，property 为声明的变量 property: function( param... ) { }, // 定义方法 generator: function*( param... ) { }, // 定义生成器 property: async function( param... ) { }, // 定义异步方法 generator: async function*( param... ) { } // 定义异步成生器 get property: function( ) { }, // getter set property: function( value ) { } // setter # 命名函数，property 为函数名 property( param... ) { }, // 定义方法 *generator( param... ) { }, // 定义生成器 async property( param... ) { }, // 定义异步方法 async* generator( param... ) { } // 定义异步成生器 get property( ) { }, // getter set property( value ) { } // setter # 箭头函数 property: ( value ) => { } } # 一般方法 var obj = { g: function( ) { }, // 完整写法 g( ){ } // 简写 } # 生成器方法 var obj = { g: function*( ) { yield 0 }, // 完整写法 *g( ){ yield 0 } // 简写 } # 异步生成器方法 var obj = { g: async function*( ) { yield 0; yield 1 }, // 完整写法 async* g( ){ yield 0; yield 1 } // 简写 } # 异步 var obj = { g: async function( ) { await some_promise }, // 完整写法 async g( ){ await some_promise } // 简写 } # 计算的属性名 var obj = { [ 'g'+1 ] : 0, [ 'g'+2 ] : function(){ }, [ 'g'+3 ]( ){ }, } var a = g1 g2( ) g3( ) # 方法定义不是构造函数 var o = new obj.method( ) // error
