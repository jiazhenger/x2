CPU.+Mac OS X/) const isMobile = !!u.match(/AppleWebKit.*Mobile.*/) if(isAndroid){ opt.onAndroid && opt.onAndroid( ) }else if(isIos){ opt.onIos && opt.onIos( ) } if(isMobile){ opt.onMobile && opt.onMobile( ) }else{ opt.onPc && opt.onPc( ) } return { isAndroid, isIos, isMobile } }
