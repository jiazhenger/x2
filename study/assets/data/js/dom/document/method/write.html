# 用法 向文档写入 HTML 表达式或 JavaScript 代码 # 语法 document.write(exp1, exp2, exp3, ...) # 参数 exp: 必需。文本字符串、html字符串、js 代码字符串 # 示例 # 写入 html document.write('<b>Hello World</b>') document.write('<b>Hello'</b>, '<b>World</b>') # 写入 js document.write('<script>alert(0)<\/script>') document.write('<b>Hello World</b>','<script>alert(0)<\/script>') # 依次写入，不会覆盖 document.write('<b>Hello'</b>) document.write('<b>World'</b>)
