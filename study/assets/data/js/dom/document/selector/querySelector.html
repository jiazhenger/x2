# 用法 返回文档中匹配指定 CSS 选择器的一个元素 # 语法 const el = document.querySelector(selectors) const el = document.querySelector(selectors).querySelector(selectors) // 子选器 # 参数 querySelector: 必需。字符串。指定一个或多个匹配元素的 CSS 选择器。 可以使用它们的 id, 类, 类型, 属性, 属性值等来选取元素。对于多个选择器，使用逗号隔开，返回一个匹配的元素 # 兼容性 ie8 + # 示例 const s = document.querySelector('div') // html 标签名 const s = document.querySelector('#id') // id const s = document.querySelector('.class') // className const s = document.querySelector('p.class') // css 限制选择器 const s = document.querySelector('a[target]') // css 属性选择器
