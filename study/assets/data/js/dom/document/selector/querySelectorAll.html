# 用法 返回文档中匹配指定 CSS 选择器的所有元素集合 # 语法 const nodeList = document.querySelectorAll(selectors) # 参数 selectors: 必需。字符串。指定一个或多个匹配元素的 CSS 选择器。 可以使用它们的 id, 类, 类型, 属性, 属性值等来选取元素。对于多个选择器，使用逗号隔开，返回一个匹配的元素 # 返回值 NodeList 对象 # 兼容性 ie8 + # 将 NodeList 转为数组 const array = Array.prototype.slice.call(nodeList,0) const array = [ ].prototype.slice.call(nodeList,0) # 示例 const nodeList = document.querySelectorAll('div') // html 标签名 const nodeList = document.querySelectorAll('#id') // id const nodeList = document.querySelectorAll('.class') // className const nodeList = document.querySelectorAll('p.class') // css 限制选择器 const nodeList = document.querySelectorAll('a[target]') // css 属性选择器
