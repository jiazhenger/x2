# 用法 创建文本节点 # 语法 const textNode = document.createTextNode(text) # 参数 text: 必需。字符串。文本节点的文本 # 示例 const textNode = document.createTextNode('文本内容') // 创建一个文本节点 document.body.appendChild(textNode ) // => 将文本节点 textNode 放入 body：<body>文本内容</body> # 示例 const node = document.createElement('div') // 创建一个元素节点 const textNode = document.createTextNode('文本内容') // 创建一个文本节点 node.appendChild(textNode ) // => 将文本节点 textNode 放入 node document.body.appendChild(node) // => 将创建的 div 放入 body 最后
