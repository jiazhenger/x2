# 用法 通过指定名称创建一个元素 # 语法 const node = document.createElement(nodename) # 参数 nodename: 必需。是创建元素的名称：html 标签元素 # 示例 const node = document.createElement('div') // <div></div> node.innerHTML = ' ' node.style.color = 'red' node.id = 'box' node.className = 'wraper' ... document.body.appendChild(node) // => 将创建的 div 放入 body 最后
