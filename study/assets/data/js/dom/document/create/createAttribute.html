# 用法 创建一个指定名称的属性，并返回 Attr 对象属性 # 语法 const attr = document.createAttribute(attributename) # 参数 attributename: 必需。要创建的属性名称 # 示例 const attr = document.createAttribute('class') // 创建一个属性名 class attr.nodeValue = 'red' // 设置属性名 class 的值为 'red' document.body.setAttributeNode(attr) === document.body.attributes.setNamedItem(attr) // => <div class='red'></div> # 示例 const node = document.createElement('div') // 创建一个元素节点 const attr = document.createAttribute('class') // 创建一个属性名 class attr.nodeValue = 'red' // 设置属性名 class 的值为 'red' node.setAttributeNode(attr) === node.attributes.setNamedItem(attr) // 将创建的属性添加到指定节点上去 class='red' document.body.appendChild(node) // => 将创建的 div 放入 body 最后
