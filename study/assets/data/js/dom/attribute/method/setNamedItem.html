# 用法 向 nodeMap 添加指定的节点 # 语法 namednodemap.setNamedItem(node) # 参数 node: 必需。Node 对象。需要在 namedNodeMap 集合中添加/替换的节点 # 示例 const attr = document.createAttribute('class') // 创建一个属性名 class attr.nodeValue = 'red' // 设置属性名 class 的值为 'red' document.body.attributes.setNamedItem(attr) // 将创建的属性添加到指定节点上去 class='red'
