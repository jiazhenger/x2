# 用法 当 readyState 属性改变时触发该事件 # readyState 语法 const status = window.document.readyState # readyState 值 uninitialized : 还未开始载入 loading : 载入中 interactive : 已加载，文档与用户可以开始交互 complete : 载入完成 # onreadystatechange 语法 document.onreadystatechange = function( e ){ const readyState = document.readyState const readyState = this.readyState } document.addEventListener('readystatechange', e => { const readyState = document.readyState const readyState = this.readyState })
