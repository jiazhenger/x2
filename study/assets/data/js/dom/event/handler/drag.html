# 第 1 步：设置 draggable = true <p id='drag' draggable='true'>拖动我</p><div id='target'></div> # 第 2 步：dragstart document.querySelector('#drag').addEventListener('dragstart', e =>{ e.dataTransfer.setData('pid', e.target.id) // 通过setData方法向dataTransfer对象添加一个属性名 pid，值为拖动元素的ID }) # 第 3 步：drop const $target = document.querySelector('#target') $target.addEventListener('drop', e =>{ const data = e.dataTransfer.getData('pid') // 获取到拖动元素的id const p = document.getElementById(data) // 通过id获取到拖动元素 $target.appendChild(p) // 把拖动数据添加到目标元素 }) # 第 4 步：dragover document.body.addEventListener('dragover', e => { e.preventDefault( ) // 阻止默认动作 })
