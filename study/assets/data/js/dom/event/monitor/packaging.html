# 添加事件 const bind = function(el, event, fn){ if (el.addEventListener) el.addEventListener(event, fn, false) else if (el.attachEvent) el.attachEvent('on' + event, fn) else if (el.click) el['on' + event] = fn else throw new Error('your browser not suport event') } # 移除事件 const unbind = function(el, event, fn){ if (el.removeEventListener) el.removeEventListener(event, fn, false) else if (el.detachEvent) el.detachEvent('on' + event, fn) else if (el.click) el['on' + event] = fn else throw new Error('your browser not suport event') }
