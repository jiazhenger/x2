# 用法 用于移除由 addEventListener( ) 方法添加的事件句柄 # 语法 window.removeEventListener(event, function, useCapture) document.removeEventListener(event, function, useCapture) document.body.removeEventListener(event, function, useCapture) document.getElementById( ).removeEventListener(event, function, useCapture) document.querySelector( ).removeEventListener(event, function, useCapture) # 参数 event : 必需。字符串，指定事件名。事件句柄没有 on function : 必需。指定要事件触发时执行的函数 useCapture : 可选。布尔值，指定事件是否在捕获或冒泡阶段执行， 默认 false true：事件句柄在捕获阶段执行 false：事件句柄在冒泡阶段执行，默认 注意：如果添加两次事件句柄，一次在捕获阶段，一次在冒泡阶段，你必须单独移除该事件 # 兼容性 ie9 + # 示例 const fn = function( ){ } document.body.addEventListener('click', fn, false) document.body.removeEventListener('click', fn, false)
