# 设置单个样式 # 追加或覆盖某个样式 el.style.color = 'red' el.style.fontSize = '20px' el.style.['font-size'] = '20px' # 追加或覆盖某个样式，并可设置优先级 el.style.setProperty('font-size','20px') el.style.setProperty('font-size','20px','important') // 在样式后加 !important # 设置多个样式 # 重写整个样式 el.style.cssText = 'color:red;font-size:20px' # 设置 style 属性 el.setAttribute('style','color:red;font-size:20px')
