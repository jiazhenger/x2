# 用法 返回 HTMLCollection 对象中指定索引位置的元素 # 语法 const node = document.getElementsByTagName('p').item(index) const node = document.getElementsByTagName('p')[index] # 参数 index: 必需。元素的索引值。起始位置为 0，最后一位元素索引为 HTMLCollection.length-1，index 需要在这个范围内 # 示例 const node document.getElementsByTagName('p').item(0) const node ddocument.getElementsByTagName('p')[0]
