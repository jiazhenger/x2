# 用法 用来表示上传的进度.这个对象是不透明的，但是作为一个XMLHttpRequestEventTarget、可以通过对其绑定事件来追踪它的进度 # 语法 const upload = xhr.upload # 绑定在上传对象上的事件监听器 xhr.upload.onloadstart = funciton( ){ } // 获取开始 xhr.upload.onload = funciton( ){ } // 获取成功 xhr.upload.onloadend = funciton( ){ } // 获取完成(不论成功与否) xhr.upload.onprogress = funciton( ){ } // 数据传输进行中 xhr.upload.ontimeout = funciton( ){ } // 获取操作在用户规定的时间内未完成 xhr.upload.onabort = funciton( ){ } // 获取操作终止 xhr.upload.onerror = funciton( ){ } // 获取失败
