# 用法 提供了一种简单，合理的方式来跨网络异步获取资源 # 语法 fetch(input,init).then(response => { }) # 参数 input : 必需。一个 USVString 字符串，包含要获取资源的网址。一些浏览器会接受 blob: 和 data: 作为方案 init : 可选。一个配置项对象，包括所有对请求的设置 # 示例 fetch('/api/search',{ method: 'POST' }).then(response => { }) # input 值 # url 地址 fetch('/api/search') # img 地址 fetch('./tortoise.png'') # Blob const req = new Request('flowers.jpg') fetch(req)
