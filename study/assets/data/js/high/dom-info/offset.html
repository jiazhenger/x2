# 相对偏移 const offsetLeft = id.offsetLeft // 上边框外边缘距离最近的已定位父级（默认 body）的距离。滚动条不影响 const offsetTop = id.offsetTop // 上边框外边缘距离最近的已定位父级（默认 body）的距离。滚动条不影响 const offsetParent = id.offsetParent // 获取已定位父级，默认 body # 绝对偏移 const getOffset = function(el){ let offsetTop = el.offsetTop let offsetLeft = el.offsetLeft let parent = el.offsetParent while(parent){ offsetTop += parent.offsetTop offsetLeft += parent.offsetLeft parent = parent.offsetParent } return { offsetTop, offsetLeft} }
