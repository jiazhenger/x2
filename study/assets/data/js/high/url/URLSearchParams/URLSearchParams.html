# 用法 处理 URL 的查询字符串 # 语法 const param = new URLSearchParams(init) # 参数 init : 可选。一个 USVString 实例，一个 URLSearchParams 实例，一个 USVString，或者一个包含 USVString 的记录。注意使用一个 URLSearchParams 实例作为参数已经被弃用了，以后的浏览器将会只使用一个 USVString 作为参数 # 用法 const param = new URLSearchParams('?a=0&b=1&c=2') // 可选参数 const param = new URLSearchParams('a=0&b=1&c=2') const param = new URLSearchParams('location.search') const param = new URLSearchParams(( new URL('https://example.com?foo=1&bar=2') ).search) const param = new URLSearchParams(( new URL(window.location) ).search) const param = new URLSearchParams([ ['a', '10'], ['b', 2] ])
