# 动态加载 js 文件 const loadJs = function( src ){ var node = document.createElement('script'); node.type = 'text/javasctipt'; node.async = true; node.src = src + '.js'; document.querySelector('head').appendChild(node) } # 使用 loadJs('./jquery') # 动态加载 css 文件 const loadCss = function( src ){ var node = document.createElement('link'); node.rel = 'stylesheet'; node.href = src + '.css'; document.querySelector('head').appendChild(node) } # 使用 loadCss('./style')
