# 用法 信息分组 # 语法 console.group(label) // 在控制台创建一个信息分组 console.groupCollapsed(label) // 创建一个折叠分组信息 console.groupEnd( ) // 用于结束分组标签 # 参数 label: 可选。分组标签 # 示例 console.log('aaaa') console.group('第一组') console.log('bbbb') console.groupEnd( ) console.log('cccc') # 示例 console.log('aaaa') console.groupCollapsed('第一组') console.log('bbbb') console.groupEnd( ) console.log('cccc')
