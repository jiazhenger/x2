/** * 有状态 Widget * 1、控件的状态、文字等属性存在变化的可能 * 2、可传参、可接收参数 * 3、有生命周期 */ # 声明一个有状态的 widget class Page extends StatefulWidget { # 声明变量 final String title; # 构造函数传参 Page({ Key key, this.title }) : super(key: key); # 创建界面 @override _PageState createState( ) => _PageState( ); } # 创建界面 class _PageState extends State<Page> { # 创建 UI @override Widget build(BuildContext context) { return Scaffold( body:Text(widget.title) // 接收 Page(title:'') 传递的参数 ); } }
