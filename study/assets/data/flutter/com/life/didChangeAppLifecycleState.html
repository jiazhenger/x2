/** * 监听生命周期 * 1、resumed：界面可见 * 2、inactive：界面退到后台或弹出对话框情况下， 即失去了焦点但仍可以执行drawframe回调 * 3、paused：应用挂起，比如退到后台，失去了焦点且不会收到drawframe回调 * 4、suspending： IOS 中没用，Android 里就是挂起，不会再执行drawframe回调 */ class _PageState extends State<Page>{ # 初次打开 widget 时，不执行 AppLifecycleState 的回调 void didChangeAppLifecycleState( state){ if( state == 'resumed' ){ 从后台到前台 }else if( state == 'inactive' ){ // 按 home 键或 Power 键 }else if( state == 'paused' ){ // back 键退出应用 }else if( state == 'suspending' ){ // IOS 中没用，Android 里就是挂起 } } }
