# pubspec.yaml 在本地字义字体 flutter: fonts: - family: Raleway // 字体名称 fonts: - asset: fonts/Raleway-Regular.ttf // 导入字体 - asset: fonts/Raleway-Medium.ttf # 将包中的字体定义到本地 // 放置位置：/my_package/lib/fonts/Raleway-Medium.ttf - asset: packages/my_package/fonts/Raleway-Medium.ttf // lib 不能放入路径中 # 加样式 weight: 500 // 加粗 style: italic // 倾斜 - asset: assets/fonts/Raleway-SemiBold.ttf weight: 600 // 加粗 - family: Schyler // 字体名称 fonts: - asset: fonts/Schyler-Regular.ttf - asset: fonts/Schyler-Italic.ttf style: italic // 倾斜 # 使用在本地字义的字体 const TextStyle(fontFamily: 'Raleway') const TextStyle(fontFamily: 'Schyler') # 使用在包中字义的字体 const TextStyle(fontFamily: 'Raleway', package: 'font_package') // 要使用包中定义的字体族，必须提供包参数
